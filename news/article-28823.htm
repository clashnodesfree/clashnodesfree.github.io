添加 tomcat7 插件

<plugins><!--tomcat插件，启动当前web应用--><plugin><groupId>org.apache.tomcat.maven</groupId><artifactId>tomcat7-maven-plugin</artifactId><version>2.1</version><configuration><!--修改tomcat启动的端口号--><port>80</port><!--修改tomcat部署当前项目虚拟目录（项目的访问跟路径）--><path>/</path></configuration></plugin></plugins>

手动输入 tomcat7:run 命令运行项目

<mirror><id>nexus-aliyun</id><mirrorOf>central</mirrorOf><name>Nexus aliyun</name><url>http://maven.aliyun.com/nexus/content/groups/public</url></mirror>
